ALTER PROCEDURE [dbo].[GetUserTasks]
    @UserId UNIQUEIDENTIFIER,       
    @Status Varchar(25) = NULL   
AS
BEGIN

    SET NOCOUNT ON;

    SELECT 
        Id,
        UserId,
        Title,
        Status,
        CreatedAt
    FROM Tasks
    WHERE UserId = @UserId
        AND (@Status = '0' OR Status = @Status)
    ORDER BY CreatedAt DESC;
END
